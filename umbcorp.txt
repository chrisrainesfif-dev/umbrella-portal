<script>
const FILE_ID = "1Yusz2TFim3OlvVzswOe_6-cDYur2VX-u";
const PDF_URL = `https://drive.google.com/file/d/${FILE_ID}/preview`;

const openBtn = document.getElementById("openBtn");
const accessOverlay = document.getElementById("accessOverlay");
const pdfOverlay = document.getElementById("pdfOverlay");
const bar = document.getElementById("bar");
const pdfFrame = document.getElementById("pdfFrame");
const timestamp = document.getElementById("timestamp");

function formatTime(){
  const d = new Date();
  return d.toISOString().replace("T"," ").split(".")[0] + " UTC";
}

openBtn.addEventListener("click", () => {

  timestamp.textContent = "Access Logged: " + formatTime();
  bar.style.width = "0%";
  accessOverlay.style.display = "flex";

  let width = 0;
  const interval = setInterval(() => {
    width += 5;
    bar.style.width = width + "%";

    if(width >= 100){
      clearInterval(interval);
      setTimeout(()=>{
        accessOverlay.style.display = "none";
        pdfOverlay.style.display = "flex";
        pdfFrame.src = PDF_URL;
      },300);
    }
  },60);
});

</script>
